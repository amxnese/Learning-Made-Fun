3. What is the output of the following switch statement?

      
        let fruit = 'apple';
        switch (fruit) {
          case 'banana':
            console.log('Yellow fruit');
            break;
          case 'apple':
            console.log('Round fruit');
            break;
          default:
            console.log('Unknown fruit');
        }
