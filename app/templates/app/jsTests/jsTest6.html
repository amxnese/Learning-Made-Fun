6. What is recursion in JavaScript?

7. Consider the following recursive function. What is its purpose?

function func(n) {
  if (n === 1) {
    return 1;
  } else {
    return n * func(n - 1);
  }
}
console.log(func(4));

8. What does the following recursive function output?

function fibonacci(n) {
  if (n < 2) {
      return n;
  } else {
      return fibonacci(n - 1) + fibonacci(n - 2);
  }
}
console.log(fibonacci(6));
